Git

  • git --version (Версія Гіта);
  • git --help (Можливості данної технології);
  • ls (список поточних папок);
  • sd + назва папки (переходить у потрібну папку);
  • git init (ініцалізує потрібну папку , створює системну папку .git);
  • git status (поточний статус);
  • git add + (назва файлу) (створює новий файл);
  • git rm --cached + (назва файлу) (видаляє файл);
  • clear (очищує консоль термінала);
  • git add . (додає всі нові файли у git);
  • git commit -m "(назва коміта)" (додає новий коміт);
  • файл .gitignore (створюється для того щоб помістити в нього потрібні файли які не потрібно відправляти на gti);
  • git branch (показує гілку на якою працюєш та всі існуючі гілки);
  • git branch (назва гілки) (створює нову гілку);
  • git branch -D (назва гілки) (видялає гілку);
  • git checkout (назва гілки) (переходиш на потрібну гілку);
  • git checkout -b (назва гілки) (створює нову гілку і відразу переходиш на цю гілку);
  • git merge (назва гілки) (з'єднує потрібну гілку , наприклад main з допоміжной гілкой іншого розробника);
  • git config --global user.name "(назва нового юзера якщо треба)" (якщо не змінюєш ім'я то по дефолту буде те що вказано)
  • git config --global user.email "(aналогічно дії тільки з елейлом що вказали у github)";
  • git branch -M main (змінює назву master на main тобто головної гілки);
  • git remote add origin https://github.com/... (вставляєш свою ссилку репозиторія щоб зв'язати його зі своїм локальним);
  • git push -u origin main (залити локальні зміни на відаленний сервер);
  • mkdir (назва папки) (створює нову папку);
  • git clone (шлях репозиторія);
  • cd .. (переходе на верхній рівень папок , виходе з поточної);
  • git pull (зтягує всі зміни що з'явились у проекті);
